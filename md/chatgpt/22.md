import os, shutil

folders = {"Images": [".jpg", ".png"], "Docs": [".pdf", ".docx"], "Videos": [".mp4"]}
for file in os.listdir():
    for folder, extensions in folders.items():
        if file.endswith(tuple(extensions)):
            os.makedirs(folder, exist_ok=True)
            shutil.move(file, folder)
